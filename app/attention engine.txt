| CREATE OR REPLACE FUNCTION public.get_personalized_feed_v3(p_user_id uuid, p_limit integer DEFAULT 20, p_cursor_score double precision DEFAULT 1.0, p_cursor_id text DEFAULT NULL::text)
 RETURNS TABLE(item_id text, item_type text, title text, summary text, author_id uuid, author_name text, created_at timestamp with time zone, likes integer, replies integer, engagement_score double precision, content_type text, original_id uuid, personalized_score double precision)
 LANGUAGE plpgsql
AS $function$
declare
    v_user_tags text[];
begin
    select array_agg(distinct interest_tag)
    into v_user_tags
    from public.user_interest_scores
    where user_id = p_user_id and interest_weight > 0.05;

    if v_user_tags is null then
        v_user_tags := '{}';
    end if;

    return query
    with base_data as (
        select
            f.item_id,
            f.item_type,
            f.title,
            f.summary,
            f.author_id,
            f.author_name,
            f.created_at,
            f.likes,
            f.replies,
            f.engagement_score,
            coalesce(f.content_type, f.item_type) as display_type,
            f.original_id,
            f.tags,
            f.engagement_score as base_score
        from public.community_feed_mv f
    ),
    scored as (
        select
            b.item_id,
            b.item_type,
            b.title,
            b.summary,
            b.author_id,
            b.author_name,
            b.created_at,
            b.likes,
            b.replies,
            b.engagement_score,
            b.display_type,
            b.original_id,
            b.tags,
            b.base_score,
            1.0 / (1 + extract(epoch from (now() - b.created_at)) / 3600.0) as recency_factor,
            case when coalesce(b.tags, '{}') && v_user_tags then 0.3 else 0.0 end as interest_boost,
            case
                when b.display_type = 'idea' then 0.35
                when b.display_type = 'blog' then 0.25
                when b.display_type = 'question' then 0.30
                else 0.1
            end as type_weight
        from base_data b
    ),
    final_scores as (
        select
            s.*,
            s.base_score
                + (s.base_score * s.interest_boost)
                + (s.base_score * s.recency_factor)
                + s.type_weight
            as total_score
        from scored s
    )
    select
        f.item_id,
        f.item_type,
        f.title,
        f.summary,
        f.author_id,
        f.author_name,
        f.created_at,
        f.likes,
        f.replies,
        f.engagement_score,
        f.display_type as content_type,
        f.original_id,
        f.total_score as personalized_score
    from final_scores f
    where (
        p_cursor_id is null
        or f.total_score < p_cursor_score
        or (f.total_score = p_cursor_score and f.item_id > p_cursor_id)
    )
    order by f.total_score desc, f.item_id desc
    limit p_limit;
end;
$function$
